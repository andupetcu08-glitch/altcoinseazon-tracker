async function updateDashboard() { try { const response = await fetch('data.json'); const data = await response.json(); // Actualizare Casete Superioare if (data.rotation) { const rotEl = document.getElementById('rotation'); rotEl.innerText = data.rotation + '%'; // Regula de 70% pentru culoarea verde if (data.rotation >= 70) { rotEl.style.color = '#00ff00'; } } if (data.btcd) document.getElementById('btcd').innerText = data.btcd + '%'; if (data.ethbtc) document.getElementById('ethbtc').innerText = data.ethbtc; if (data.usdtd) document.getElementById('usdtd').innerText = data.usdtd + '%'; // Fix pentru Multiplier Actual if (data.mult !== undefined) { document.getElementById('multiplier').innerText = data.mult + 'x'; } else { document.getElementById('multiplier').innerText = '0.00x'; } // Portofoliu Actual (EUR) if (data.port_eur) { document.getElementById('port_eur').innerText = 'â‚¬' + data.port_eur.toLocaleString(); } // Casete Macro if (data.vix) document.getElementById('vix').innerText = data.vix; if (data.dxy) document.getElementById('dxy').innerText = data.dxy; if (data.fng) document.getElementById('fng').innerText = data.fng; } catch (error) { console.error("Eroare la incarcarea datelor:", error); } } // Ruleaza la fiecare 30 secunde setInterval(updateDashboard, 30000); updateDashboard();
